wangEditor demo

在demo-basic.html应用的基础上，可通过传入'hideMenuConfig'，来配置你想隐藏的菜单按钮。

$('#textarea1').wangEditor({
	'hideMenuConfig': ['removeFormat', 'insertHr']   //隐藏掉“removeFormat”和“insertHr”两个按钮
});

wangEditor自带的菜单配置如下，可以从其中酌情筛选你想要隐藏的。

//默认的菜单显示配置
[
    'fontFamily', 'fontSize', '|', 
    'bold', 'underline', 'italic', '|', 
    'setHead', 'foreColor', 'backgroundColor', 'removeFormat', '|', 
    'indent', 'outdent', '|',
    'unOrderedList', 'orderedList', '|', 
    'justifyLeft', 'justifyCenter', 'justifyRight', '|', 
    'createLink', 'unLink', '|', 
    'insertHr', 'insertTable', 'webImage', 'insertSimpleCode', 'insertHighLighterCode', '|',
    'undo', 'redo'
]
